44static void handleToken(void){
45 T_token tok = nextToken();
46 int arg1, arg2;
47 int stackStatus = 0; // in case of an error: negativ number and errno set
48 switch (tok.tok) {
49 case PLUS: case MINUS: case MULT: case DIV: case SWAP:
50 // all binary operators arg1 op arg2
51 stackStatus = pop(&arg2);
52 stackStatus = stackStatus + pop(&arg1);
53 if (SWAP == tok.tok) {
54 stackStatus = push(arg2);
55 stackStatus = stackStatus + push(arg1);
56 }
57 if (stackStatus != 0) {
58 break;
59 }
60 if ((PLUS == tok.tok) && overflowAdd(arg1, arg2) ||
61 (MINUS == tok.tok) && overflowSub(arg1, arg2) ||
62 (MULT == tok.tok) && overflowMult(arg1,arg2) ||
63 (DIV == tok.tok) && overflowDiv(arg1,arg2)) {
64 errHandler("Arithmetic overflow");
65 break;
66 }
67 stackStatus = (PLUS == tok.tok) ? push(arg1 + arg2) : stackStatus;
68 stackStatus = (MINUS == tok.tok) ? push(arg1 - arg2) : stackStatus;
69 stackStatus = (MULT == tok.tok) ? push(arg1 * arg2) : stackStatus;
70 stackStatus = (DIV == tok.tok) ? push(arg1 / arg2) : stackStatus;
71 break;
72 case NUMBER: case DOUBLE: // unary operators
73 arg1 = tok.val;
74 if (DOUBLE == tok.tok){
75 stackStatus = elemAt(0, &arg1);
76 }
77 stackStatus = stackStatus + push(arg1);
78 break;
79 case CLEAR:
80 clearSystem();
81 break;
82 case OVERFLOW:
83 errHandler("Overflow during input of number");
84 break;
85 case UNEXPECTED:
86 errHandler("Internal error: unexpected input key");
87 break;
88 case PRT: case PRT_ALL:
89 clearStdout();
90 if (0 != elemAt(0,&arg1)){
91 printStdout("empty stack\n");
92 break;
93 }
94 printNumber(arg1);
95 int i = 1;
96 while((PRT_ALL == tok.tok) && (0 == elemAt(i++,&arg1))){
97 printNumber(arg1);
98 if (isBottomLine()) {
99 printStdout("Press P to continue\n");
100 while (PRT_ALL != nextToken().tok) {}
101 setNormalMode();
102 }
103 }
104 break;
105 default: // ENTER token not expected, is whitespace
106 errHandler("Internal error: unexpected token");
107 }
108 // report stack error
109 if (stackStatus != 0) {
110 errHandler("Stack overflow or stack underflow");
111 }
112}
